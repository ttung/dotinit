    set watch = (1 any any)

    umask 066

    setenv EDITOR	"emacs-nw"
    #setenv NNTPSERVER news.comcast.giganews.com
    setenv NNTPSERVER localhost:1190

    if ($?f_interactive) then
        alias	fc	finger christinewu
        alias	fy	finger ctung
        alias	idle	midle ${users}
        foreach rhost (${fastremotehosts})
            set srhost = `echo $rhost | sed 's/\..*//'`

            alias	${srhost}	ssh ${rhost}
        end
        unset srhost
        alias	csua	ssh -L 1190:agate.berkeley.edu:119 csua.berkeley.edu

        complete	finger	"p/1/(${users})/"
        complete	f	"p/1/(${users})/"

        cstat "."

        if ($?WINDOW) then

                screen -X login
                screen -X reportidle
            endif
        endif
        if (! $?prompt && $?tcsh && $TERMTYPE == screen) then
            if ($?WINDOW) then
                setenv WINDOW_NUM "-$WINDOW"
                unsetenv WINDOW
            else
                setenv WINDOW_NUM
            endif

            set prompt='%{\ek\e\\\e]0\;[%m'${WINDOW_NUM:q}']:%/>^g%}[%m'${WINDOW_NUM:q}']:%B%~%b%# '

        endif
    endif

    cstat "done\n"
