        complete	ssh	"p/1/(${remotehosts})/"

                screen -X login

        endif
        if (! $?prompt && $?tcsh && $TERMTYPE == screen) then
            if ($?WINDOW) then
                setenv WINDOW_NUM "-$WINDOW"
                unsetenv WINDOW
            else
                setenv WINDOW_NUM
            endif

            set prompt='%{\ek\e\\\e]0\;[%m'${WINDOW_NUM:q}']:%/>^g%}[%m'${WINDOW_NUM:q}']:%B%~%b%# '

            periodic
        endif
    endif

    cstat "done\n"
